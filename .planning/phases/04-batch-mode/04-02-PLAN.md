```bash
CVE_INPUT="CVE-2021-44228

  invalid-line
CVE-1999-0001"

VALID_CVES=$(printf '%s' "$CVE_INPUT" \
  | tr ',' '\n' \
  | sed 's/^[[:space:]]*//;s/[[:space:]]*$//' \
  | grep -E '^CVE-[0-9]{4}-[0-9]{4,}$' \
  || true)

MATRIX=$(printf '%s' "$VALID_CVES" \
  | jq -R -s 'split("\n") | map(select(length > 0)) | tojson')

echo "Matrix output: $MATRIX"
```
